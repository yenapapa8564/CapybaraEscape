# 벽 낙서(Graffiti) 구현 계획

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 플레이어가 스프레이 캔 아이템으로 벽에 자유 드로잉을 남길 수 있는 기능 구현

**Architecture:** 클라이언트(ClientUI)에서 오버레이 캔버스로 스트로크 데이터 수집 → GraffitiPlaced RemoteEvent로 서버에 전송 → ItemManager가 검증 후 벽 파트에 SurfaceGui 생성. 기존 smoke/trap 아이템 패턴을 그대로 따름.

**Tech Stack:** Roblox Lua (서버/클라이언트), RemoteEvent, SurfaceGui, UserInputService, RunService

### Task 5: placeGraffiti — 검증 로직

**Files:**
- Modify: `src/ServerScriptService/Modules/ItemManager.lua`

- [ ] **Step 1: placeGraffiti 검증 단계 구현**

Task 4의 스텁을 아래로 교체:

```lua
function ItemManager.placeGraffiti(player, data, mazeFolder)
    local PlayerManager = require(game.ServerScriptService.Modules.PlayerManager)
    local Constants     = require(game.ReplicatedStorage.Shared.Constants)

    -- 1. mazeFolder nil 체크
    if not mazeFolder or not mazeFolder.Parent then return end

    -- 2. 스테이지 낙서 수 제한
    if graffitiCount >= Constants.GRAFFITI_MAX_STAGE then return end

    -- 3. 데이터 기본 검증
    if type(data) ~= "table" then return end
    if type(data.strokes) ~= "table" then return end
    if #data.strokes == 0 or #data.strokes > 20 then return end
    if typeof(data.originPos)  ~= "Vector3" then return end
    if typeof(data.lookVector) ~= "Vector3" then return end

    -- 4. consumeSpray 원자적 처리 (count >= 1 확인 후 소모)
    if not PlayerManager.consumeSpray(player) then return end
    -- 이후 검증 실패 시 환불 함수
    local function refund() PlayerManager.addSpray(player) end

    -- 5. 서버측 거리 재검증 (originPos 스냅샷 기준)
    local CollectionService = game:GetService("CollectionService")
    local nearbyParts = workspace:GetPartBoundsInRadius(data.originPos, Constants.GRAFFITI_RANGE)
    local wallFound = false
    for _, p in ipairs(nearbyParts) do
        local isWall = (p.Name == "Wall" or CollectionService:HasTag(p, "MazeWall"))
        if isWall and p:IsDescendantOf(mazeFolder) then
            wallFound = true
            break
        end
    end
    if not wallFound then refund(); return end

    -- 6. 스트로크 서버 검증 및 정규화
    local validatedStrokes = {}
    for _, stroke in ipairs(data.strokes) do
        if type(stroke) ~= "table" then refund(); return end
        if type(stroke.points) ~= "table" then refund(); return end
        if #stroke.points == 0 or #stroke.points > 200 then refund(); return end
        if typeof(stroke.color) ~= "Color3" then refund(); return end

        local thickness = tonumber(stroke.thickness) or 2
        thickness = math.clamp(math.floor(thickness), 1, 3)

        local snappedColor = snapToGraffitiPalette(stroke.color)
        local cleanPoints = {}
        for _, pt in ipairs(stroke.points) do
            if type(pt) == "table" and type(pt.x) == "number" and type(pt.y) == "number" then
                table.insert(cleanPoints, {
                    x = math.clamp(pt.x, 0, 1),
                    y = math.clamp(pt.y, 0, 1),
                })
            end
        end
        if #cleanPoints < 2 then refund(); return end

        table.insert(validatedStrokes, {
            color     = snappedColor,
            thickness = thickness,
            points    = cleanPoints,
        })
    end

    -- 7. Raycast로 벽 Face 탐지
    local rayResult = workspace:Raycast(
        data.originPos,
        data.lookVector.Unit * (Constants.GRAFFITI_RANGE + 1)
    )
    if not rayResult or not rayResult.Instance then refund(); return end
    local targetPart = rayResult.Instance
    local isWallRay = (targetPart.Name == "Wall" or CollectionService:HasTag(targetPart, "MazeWall"))
    if not isWallRay or not targetPart:IsDescendantOf(mazeFolder) then
        refund(); return
    end

    -- 8. SurfaceGui 렌더링
    graffitiCount += 1
    _renderGraffiti(targetPart, rayResult.Normal, validatedStrokes, player.Name)

    -- 9. 클라이언트 HUD 업데이트 (스프레이 사용 후 남은 수 또는 버튼 숨김)
    local Events = game.ReplicatedStorage.Events
    Events.UpdateHUD:FireClient(player, {
        type  = "sprayUsed",
        count = PlayerManager.getSprayCount(player),
    })
end
```

```lua
-- 두 점 사이의 선분을 회전된 Frame으로 SurfaceGui 위에 그린다
local function drawSegment(parent, x1, y1, x2, y2, color, thickness, cs)
    local dx = x2 - x1
    local dy = y2 - y1
    local len = math.sqrt(dx*dx + dy*dy)
    if len < 0.001 then return end

    local cx = (x1 + x2) * 0.5
    local cy = (y1 + y2) * 0.5
    local angle = math.atan2(dy, dx)

    local thicknessMap = {[1]=2, [2]=4, [3]=7}  -- thickness 1~3 → px
    local px = thicknessMap[thickness] or 4

    local seg = Instance.new("Frame")
    seg.BackgroundColor3 = color
    seg.BorderSizePixel  = 0
    -- 길이: len*cs px, 두께: px
    seg.Size     = UDim2.new(0, len * cs, 0, px)
    seg.Position = UDim2.new(0, cx * cs - (len * cs) * 0.5, 0, cy * cs - px * 0.5)
    seg.Rotation = math.deg(angle)
    seg.ZIndex   = 2
    seg.Parent   = parent

    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0.5, 0)
    corner.Parent = seg
end

-- 벽 파트에 SurfaceGui를 생성하고 스트로크를 렌더링
local function _renderGraffiti(wallPart, normal, strokes, authorName)
    local Constants = require(game.ReplicatedStorage.Shared.Constants)
    local cs = Constants.GRAFFITI_CANVAS_SIZE  -- 500

    -- Face 결정: normal 벡터에서 가장 가까운 면
    local faceMap = {
        [Vector3.new( 0, 0,-1)] = Enum.NormalId.Front,
        [Vector3.new( 0, 0, 1)] = Enum.NormalId.Back,
        [Vector3.new(-1, 0, 0)] = Enum.NormalId.Left,
        [Vector3.new( 1, 0, 0)] = Enum.NormalId.Right,
        [Vector3.new( 0, 1, 0)] = Enum.NormalId.Top,
        [Vector3.new( 0,-1, 0)] = Enum.NormalId.Bottom,
    }
    local bestFace, bestDot = Enum.NormalId.Front, -math.huge
    for vec, face in pairs(faceMap) do
        local d = normal:Dot(vec)
        if d > bestDot then bestDot, bestFace = d, face end
    end

    local sg = Instance.new("SurfaceGui")
    sg.Name          = "GraffitiGui"
    sg.Face          = bestFace
    sg.CanvasSize    = Vector2.new(cs, cs)
    sg.ResetOnSpawn  = false
    sg.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
    sg.LightInfluence = 0
    sg.Parent        = wallPart  -- 벽 파트에 직접 부착

    -- 배경 (투명)
    local bg = Instance.new("Frame")
    bg.BackgroundTransparency = 1
    bg.Size   = UDim2.new(1, 0, 1, 0)
    bg.ZIndex = 1
    bg.Parent = sg

    -- 스트로크 렌더
    for _, stroke in ipairs(strokes) do
        local pts = stroke.points
        for i = 1, #pts - 1 do
            drawSegment(bg,
                pts[i].x, pts[i].y,
                pts[i+1].x, pts[i+1].y,
                stroke.color, stroke.thickness, cs)
        end
    end

    -- 작성자 서명 (우하단)
    local sig = Instance.new("TextLabel")
    sig.Text                 = authorName
    sig.Font                 = Enum.Font.Gotham
    sig.TextSize             = 14
    sig.TextColor3           = Color3.fromRGB(220, 220, 220)
    sig.TextStrokeColor3     = Color3.fromRGB(0, 0, 0)
    sig.TextStrokeTransparency = 0.4
    sig.BackgroundTransparency = 1
    sig.AnchorPoint          = Vector2.new(1, 1)
    sig.Position             = UDim2.new(1, -6, 1, -4)
    sig.Size                 = UDim2.new(0, 120, 0, 20)
    sig.TextXAlignment       = Enum.TextXAlignment.Right
    sig.ZIndex               = 3
    sig.Parent               = bg
end
```

local function addSegmentVisual(x1, y1, x2, y2, color, thickness)
    local dx = x2 - x1
    local dy = y2 - y1
    local len = math.sqrt(dx*dx + dy*dy)
    if len < 2 then return end

    local cx = (x1 + x2) * 0.5
    local cy = (y1 + y2) * 0.5
    local angle = math.deg(math.atan2(dy, dx))
    local thickPx = ({[1]=2, [2]=4, [3]=7})[thickness] or 4

    local seg = Instance.new("Frame")
    seg.BackgroundColor3 = color
    seg.BorderSizePixel  = 0
    seg.Size             = UDim2.new(0, len, 0, thickPx)
    seg.Position         = UDim2.new(0, cx - len*0.5, 0, cy - thickPx*0.5)
    seg.Rotation         = angle
    seg.ZIndex           = 35
    seg.Parent           = canvasArea
    Instance.new("UICorner", seg).CornerRadius = UDim.new(0.5, 0)
    return seg
end
